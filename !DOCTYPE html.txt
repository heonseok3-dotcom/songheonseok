    <meta charset="UTF-8">
    <title>연결리스트 오목 게임</title>
    <style>
        body { display: flex; flex-direction: column; align-items: center; background-color: #f0f0f0; font-family: sans-serif; }
        canvas { background-color: #e3a05b; box-shadow: 0 0 20px rgba(0,0,0,0.3); cursor: crosshair; }
        .status { margin: 20px; font-size: 1.5rem; font-weight: bold; }
        button { padding: 10px 20px; font-size: 1rem; cursor: pointer; background: #333; color: white; border: none; border-radius: 5px; }
    </style>

<body>

    <div class="status" id="status">흑색(Black) 차례</div>
    <canvas id="board" width="600" height="600"></canvas>
    <br>
    <button onclick="resetGame()">게임 초기화</button>

    <script>
        const canvas = document.getElementById('board');
        const ctx = canvas.getContext('2d');
        const statusEl = document.getElementById('status');
        
        const size = 15; // 15x15 판
        const padding = 40;
        const cell = (canvas.width - padding * 2) / (size - 1);
        
        let turn = 'black'; // 'black' or 'white'
        let gameOver = false;

        // --- 1. 연결리스트 노드 클래스 정의 ---
        class Node {
            constructor(x, y) {
                this.x = x;
                this.y = y;
                this.color = null; // null, 'black', 'white'
                // 8방향 연결 (우, 우하, 하, 좌하) - 반대 방향은 탐색 시 유연하게 처리
                this.neighbors = {
                    right: null, bottom: null, bottomRight: null, bottomLeft: null
                };
            }
        }

        let nodes = [];

        // --- 2. 바둑판 초기화 (노드 생성 및 연결) ---
        function initBoard() {
            nodes = Array.from(Array(size), () => Array(size));
            
            // 노드 생성
            for (let y = 0; y < size; y++) {
                for (let x = 0; x < size; x++) {
                    nodes[y][x] = new Node(x, y);
                }
            }

            // 노드 간 연결 (Linked List 구조 형성)
            for (let y = 0; y < size; y++) {
                for (let x = 0; x < size; x++) {
                    if (x < size - 1) nodes[y][x].neighbors.right = nodes[y][x + 1];
                    if (y < size - 1) nodes[y][x].neighbors.bottom = nodes[y + 1][x];
                    if (x < size - 1 && y < size - 1) nodes[y][x].neighbors.bottomRight = nodes[y + 1][x + 1];
                    if (x > 0 && y < size - 1) nodes[y][x].neighbors.bottomLeft = nodes[y + 1][x - 1];
                }
            }
        }

        // --- 3. 승리 조건 체크 (연결리스트 탐색) ---
        function checkWin(node) {
            const directions = [
                { key: 'right', dx: 1, dy: 0 },
                { key: 'bottom', dx: 0, dy: 1 },
                { key: 'bottomRight', dx: 1, dy: 1 },
                { key: 'bottomLeft', dx: -1, dy: 1 }
            ];

            for (let d of directions) {
                let count = 1;
                
                // 정방향 탐색 (예: 오른쪽)
                let curr = getNeighbor(node, d.dx, d.dy);
                while (curr && curr.color === node.color) {
                    count++;
                    curr = getNeighbor(curr, d.dx, d.dy);
                }

                // 역방향 탐색 (예: 왼쪽)
                curr = getNeighbor(node, -d.dx, -d.dy);
                while (curr && curr.color === node.color) {
                    count++;
                    curr = getNeighbor(curr, -d.dx, -d.dy);
                }

                if (count >= 5) return true;
            }
            return false;
        }

        // 특정 좌표의 이웃 노드 가져오기 유틸리티
        function getNeighbor(node, dx, dy) {
            const nx = node.x + dx;
            const ny = node.y + dy;
            if (nx >= 0 && nx < size && ny >= 0 && ny < size) {
                return nodes[ny][nx];
            }
            return null;
        }

        // --- 4. 렌더링 및 이벤트 ---
        function drawBoard() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            ctx.strokeStyle = '#000';
            for (let i = 0; i < size; i++) {
                // 가로줄
                ctx.beginPath();
                ctx.moveTo(padding, padding + i * cell);
                ctx.lineTo(canvas.width - padding, padding + i * cell);
                ctx.stroke();
                // 세로줄
                ctx.beginPath();
                ctx.moveTo(padding + i * cell, padding);
                ctx.lineTo(padding + i * cell, canvas.height - padding);
                ctx.stroke();
            }
            
            // 돌 그리기
            for (let y = 0; y < size; y++) {
                for (let x = 0; x < size; x++) {
                    if (nodes[y][x].color) {
                        ctx.beginPath();
                        ctx.arc(padding + x * cell, padding + y * cell, cell/2.5, 0, Math.PI * 2);
                        ctx.fillStyle = nodes[y][x].color;
                        ctx.fill();
                        ctx.stroke();
                    }
                }
            }
        }

        canvas.addEventListener('click', (e) => {
            if (gameOver) return;

            const rect = canvas.getBoundingClientRect();
            const mouseX = e.clientX - rect.left;
            const mouseY = e.clientY - rect.top;

            const x = Math.round((mouseX - padding) / cell);
            const y = Math.round((mouseY - padding) / cell);

            if (x >= 0 && x < size && y >= 0 && y < size && !nodes[y][x].color) {
                nodes[y][x].color = turn;
                drawBoard();

                if (checkWin(nodes[y][x])) {
                    statusEl.innerText = `${turn === 'black' ? '흑색' : '백색'} 승리!`;
                    gameOver = true;
                    return;
                }

                turn = turn === 'black' ? 'white' : 'black';
                statusEl.innerText = `${turn === 'black' ? '흑색' : '백색'} 차례`;
            }
        });

        function resetGame() {
            turn = 'black';
            gameOver = false;
            statusEl.innerText = "흑색(Black) 차례";
            initBoard();
            drawBoard();
        }

        initBoard();
        drawBoard();
    </script>
